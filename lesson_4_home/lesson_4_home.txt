function maxNum($arr){
    $a = 0;
    $num = '';

    for ($i = 0; $i < count($arr); $i++){
        if ($arr[$i] > $a){
            $a = $arr[$i];
            $num = $i;
        }
    }
    unset($arr["$num"]);
    return $arr;
}

$s = maxNum($mas);
print_r($s);

function maxStr($arr){
    $a = 0;
    $num = 0;
    for ($i = 0; $i < count($arr); $i++){
        if (strlen($arr[$i]) > $a){
            $a = strlen($arr[$i]);
            $num = $i;
        }
    }
    unset($arr["$num"]);
    return $arr;
}
